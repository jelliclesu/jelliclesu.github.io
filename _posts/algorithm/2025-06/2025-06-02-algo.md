import java.util.Scanner;

public class Main {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();

    boolean[] isClosed = new boolean[n + 1];
    long count = n;
    for (int i = 2; i <= n; i++) {
      for (int j = i; j <= n; j += i) {
        if (isClosed[j]) {
          isClosed[j] = false;
          count++;
        } else {
          isClosed[j] = true;
          count--;
        }
      }
    }
    System.out.println(count);
  }
}

import java.util.Scanner;

public class Main {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    long n = sc.nextLong();
    long answer = 0;

    for (int i = 1; i <= n; i++) {
      long count = 0;
      for (int j = 1; j <= Math.sqrt(n); j++) {
        if (i % j == 0) {
          count++;
          if (j != i / j) count++;
        }
      }
      if (count % 2 != 0) {
        answer++;
      }
    }
    System.out.println(answer);
  }
}
