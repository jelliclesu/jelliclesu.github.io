class Solution {
  public String solution(int[] numbers) {

    String[] strNums = Arrays.stream(numbers).mapToObj(String::valueOf).toArray(String[]::new);

    Arrays.sort(strNums, (a, b) -> (b + a).compareTo(a + b));

    return String.join("", strNums);
  }
}
