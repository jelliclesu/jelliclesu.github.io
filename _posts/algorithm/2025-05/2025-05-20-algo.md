import java.util.*;

public class SWEA_1244_1 {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int[] num = Arrays.stream(sc.next().split(""))
      .mapToInt(Integer::parseInt)
      .toArray();
    int count = sc.nextInt();

    int swap = 0;
    String now = "";
    String prev = "";
    for (int i = 0; i < num.length; i++) {
      if (swap == 0 && i == num.length - 1) {
        int temp = num[i];
        num[i] = num[i - 1];
        num[i - 1] = temp;
        break;
      }
      int max = 0;
      int idx = 0;
      for (int j = i; j < num.length; j++) {
        max = Math.max(max, num[j]);
        if (max == num[j]) {
          idx = j;
        }
      }
      if (idx == i) continue;
      int temp = num[idx];
      num[idx] = num[i];
      num[i] = temp;
      swap++;
      String[] nowArr = Arrays.stream(num)
        .mapToObj(String::valueOf)
        .toArray(String[]::new);
      now = String.join("", nowArr);
      if (swap == count || prev.equals(now)) break;
      prev = now;
    }
    System.out.println(now);
  }
}

package com.prac.algorithm;

import java.util.HashSet;
import java.util.Scanner;
import java.util.Set;

public class SWEA_1244_2 {
  static int max = 0;
  static Set<String>[] visited;

  public static void main(String[] args) {
    Scanner sc= new Scanner(System.in);

    char[] num = sc.next().toCharArray();
    int count = sc.nextInt();

    max = 0;
    visited = new HashSet[count + 1];   // ê° ê¹Šì´ë§ˆë‹¤ ë°©ë¬¸í•œ ë¬¸ìì—´ ì €ì¥ìš©
    for (int i = 0; i <= count; i++) {

    dfs(num, count, 0);

    System.out.println(max);
  }

  private static void dfs(char[] num, int count, int depth) {
    if (depth == count) {
      int n = Integer.parseInt(new String(num));
      max = Math.max(max, n);
      return;
    }

    String now = new String(num);
    if (visited[depth].contains(now)) return;   // ì´ë¯¸ depth ì— ìˆëŠ” ê°’ì´ë©´ íƒìƒ‰ í•„ìš” ì—†ìŒ
    visited[depth].add(now);

      for (int j = i + 1; j < num.length; j++) {
        swap(num, i, j);
        dfs(num, count, depth + 1);
        swap(num, i, j);    // ë°±íŠ¸ë˜í‚¹
      }
    }
  }

  private static void swap(char[] num, int i, int j) {
    char temp = num[i];
    num[i] = num[j];
    num[j] = temp;
  }
}
